class Solution {
public:
    std::map<std::string, int> dict = {
            {"I",1},
            {"V",5},
            {"X",10},
            {"L",50},
            {"C",100},
            {"D",500},
            {"M",1000},
            {"IV",4},
            {"IX",9},
            {"XL",40},
            {"XC",90},
            {"CD",400},
            {"CM",900}
    };
    int romanToInt(string s) {
        
        int ans = 0;
        for (int i = 0; i < s.length(); ++i) {
            if (i + 1 < s.length() && dict.find(s.substr(i, 2)) != dict.end()) {
                ans += dict[s.substr(i, 2)];
                ++i;
            } else {
                ans += dict[s.substr(i, 1)];
            }
        }
        return ans;
    }
};
