class Solution {
public:
    std::string call(const std::string& s) {
        if (s.length() <= 0) return "";
        char ch = s[0];
        int count = 1;
        std::stringstream ss;
        for (int i = 1; i < s.length(); ++i) {
            if (s[i] != ch) {
                ss << count;
                ss << ch;
                ch = s[i];
                count = 1;
            } else {
                ++count;
            }
        }
        ss << count;
        ss << ch;
        return ss.str();
    }
    
    string countAndSay(int n) {
        std::string ans[30];
        ans[0] = "1";
        for (int i = 1; i < n; ++i) {
            ans[i] = call(ans[i - 1]);
        }
        return ans[n - 1];
    }
};
