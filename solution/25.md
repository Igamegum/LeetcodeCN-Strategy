class Solution {
public:
    void reverse(ListNode* prev, ListNode* st, ListNode* ed) {
        if (st == NULL || ed == NULL || st == ed) return;
        ListNode* slow = st;
        ListNode* fast = st->next;
        ListNode* edn = ed->next;
        
        while (fast != edn ) {
            ListNode* temp = fast->next;
            fast->next = slow;
            slow = fast;
            fast = temp;
        }
        if (prev) prev->next = slow;
        st->next = edn;
    }

    ListNode* solve(ListNode* head, int K) {
        if (K < 2) return head;
        ListNode* prev = NULL;
        ListNode* st = head;
        ListNode* ed = head;
        ListNode* ans = head;
        int count = 1;
        while (ed) {
            if (count == K) {
                if (prev == NULL) ans = ed;
                ListNode* edn = ed->next;
			    reverse(prev, st, ed);
			    prev = st;
			    ed = edn;
			    st = edn;
                count = 1;
            } else {
                ed = ed->next;
                ++count;
            }
        }
        return ans;
    }
    ListNode* reverseKGroup(ListNode* head, int k) {
        return solve(head, k);
    }
};
