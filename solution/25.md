class Solution {
public:
    void reverse(ListNode* prev, ListNode* st, ListNode* ed) {
        ListNode* suff = ed->next;
        ListNode* p = prev;
        ListNode* q = st;
        while (q) {
            ListNode* temp = q->next;
            q->next = p;
            p = q;
            if (q == ed) break;
            q = temp;
        }
        prev->next = ed;
        st->next = suff;
    }

    ListNode* reverseKGroup(ListNode* head, int k) {
        ListNode* bias = new ListNode(0);
        bias->next = head;
        ListNode* prev = bias;
        ListNode* st = head;
        ListNode* ed = head;
        int cnt = 0;
        while (ed) {
            ++cnt;
            if (cnt == k) {
                ListNode* edn = ed->next;
                reverse(prev, st, ed);
                prev = st;
                st = edn;
                ed = edn;
                cnt = 0;
            } else {
                ed = ed->next;
            }
        }

        return bias->next;
    }
};
