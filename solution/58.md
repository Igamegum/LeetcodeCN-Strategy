class Solution {
public:
    int lengthOfLastWord(string s) {
        std::string str;
        for (int i = s.size() - 1; i >= 0; --i) {
            if (s[i] == ' ' && str.size()) {
                break;
            } else if (s[i] != ' ') {
                str += s[i];
            }
        }
        return str.size();
    }
};
